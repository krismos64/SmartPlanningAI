      );
 
      // Activits par jour (30 derniers jours)
      const [dailyStats] = await db.query(
        `
        SELECT DATE(timestamp) as date, COUNT(*) as count 
        FROM activities 
        WHERE timestamp > DATE_SUB(NOW(), INTERVAL 30 DAY)
        GROUP BY DATE(timestamp)
        ORDER BY date
        `
      );
 
      return {
        total: totalResult[0].total,
        byType: typeStats,
        byUser: userStats.map((user) => ({
          user_id: user.user_id,
          name:
            user.username ||
            `${user.first_name} ${user.last_name}`.trim() ||
            "Utilisateur inconnu",
          count: user.count,
        })),
        byDay: dailyStats,
      };

      console.error(
        "Erreur lors de la rcupration des statistiques d'activits:",

  /**
   * Formate une description d'activit en fonction de son type et de ses dtails
   * @param {Object} activity - Activit  formater
   * @returns {string} - Description formate
   */
  static formatActivityDescription(activity) {
    try {
      if (!activity) return "";
 
      const { type, entity_type, entity_id, description, details } = activity;
 
      // Si une description personnalise est fournie, l'utiliser directement
      if (description) {
        return description;
      }
 
      // Formater le type d'entit
      const entityName =
        {
          employee: "un employ",
          schedule: "un planning",
          vacation: "une demande de cong",
          shift: "un horaire",
          user: "un utilisateur",
        }[entity_type] || entity_type;
 
      // Formater le type d'action
      const actionType =
        {
          create: "a cr",
          update: "a modifi",
          delete: "a supprim",
          approve: "a approuv",
          reject: "a rejet",
          vacation_status_update: "a mis  jour le statut de",
        }[type] || type;
 
      // Construire la description
      let formattedDescription = `Un utilisateur ${actionType} ${entityName}`;
 
      // Ajouter des dtails si disponibles
      if (details) {
        let parsedDetails;
        try {
          parsedDetails =
            typeof details === "string" ? JSON.parse(details) : details;
        } catch (e) {
          parsedDetails = details;
        }
 
        // Cas spcial pour les modifications de solde d'heures
        if (
          entity_type === "employee" &&
          parsedDetails.action &&
          (parsedDetails.action === "Ajout d'heures" ||
            parsedDetails.action === "Soustraction d'heures")
        ) {
          const userName = parsedDetails.userName || "Un utilisateur";
          const employeeName =
            parsedDetails.employeeName || `Employ #${entity_id}`;
          const hours = parsedDetails.hours || "?";
          const action =
            parsedDetails.action === "Ajout d'heures" ? "ajout" : "soustrait";
 
          return `${userName} a ${action} ${hours}h au solde d'heures de ${employeeName}`;
        }
 
        // Cas spcial pour les demandes de congs
        if (entity_type === "vacation") {
          const userName =
            activity.first_name && activity.last_name
              ? `${activity.first_name} ${activity.last_name}`
              : "Un utilisateur";
 
          // Pour la cration de congs
          if (
            type === "create" &&
            parsedDetails.employee_id &&
            parsedDetails.start_date &&
            parsedDetails.end_date
          ) {
            const employeeName =
              parsedDetails.employee_name ||
              `Employ #${parsedDetails.employee_id}`;
            const typeConge = parsedDetails.type || "non spcifi";
            const startDate = new Date(
              parsedDetails.start_date
            ).toLocaleDateString("fr-FR");
            const endDate = new Date(parsedDetails.end_date).toLocaleDateString(
              "fr-FR"
            );
 
            return `${userName} a cr une demande de cong ${typeConge} pour ${employeeName} du ${startDate} au ${endDate}`;
          }
 
          // Pour la mise  jour du statut des congs
          if (
            type === "vacation_status_update" &&
            parsedDetails.previous_status &&
            parsedDetails.new_status
          ) {
            const previousStatus = parsedDetails.previous_status;
            const newStatus = parsedDetails.new_status;
 
            let statusText = "";
            if (newStatus === "approved") statusText = "approuv";
            else if (newStatus === "rejected") statusText = "rejet";
            else if (newStatus === "pending") statusText = "remis en attente";
 
            return `${userName} a ${statusText} la demande de cong #${entity_id}`;
          }
 
          // Pour la mise  jour gnrale des congs
          if (type === "update") {
            return `${userName} a modifi la demande de cong #${entity_id}`;
          }
 
          // Pour la suppression des congs
          if (type === "delete") {
            return `${userName} a supprim la demande de cong #${entity_id}`;
          }
        }
 
        // Pour les autres types d'activits
        if (typeof parsedDetails === "string") {
          formattedDescription += ` : ${parsedDetails}`;
        } else if (typeof parsedDetails === "object") {
          if (parsedDetails.employeeName && entity_type === "employee") {
            formattedDescription += ` : ${parsedDetails.employeeName}`;
          }
        }
      }
 
      return formattedDescription;
    } catch (error) {
      console.error("Erreur lors du formatage de la description:", error);
      return "Activit inconnue";
    }
  }
